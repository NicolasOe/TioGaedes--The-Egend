;-------------------------------------------------------
read_entry PROC  USES ebx
; Esse procedimento usa um loop para checar uma entrada do teclado
; Saída: al
;-------------------------------------------------------
.code
mov al,0

;loop para receber input do teclado 
P031:	
	Call ReadChar	;le uma entrada do teclado

	cmp al,0		;checa se houve uma entrada 
	jne P03Saida

jmp P031

P03Saida:
ret
read_entry ENDP
